Copy-- ── migration: 27_restrict_update_with_check.sql ─────────────────────────────
-- Restringe WITH CHECK no UPDATE: collaborators não podem alterar
-- owner_id, title, is_shared, ou share_token.

DROP POLICY IF EXISTS "diagrams_update_policy" ON public.diagrams;

CREATE POLICY "diagrams_update_policy"
  ON public.diagrams FOR UPDATE
  TO authenticated
  USING (
    deleted_at IS NULL AND (
      owner_id = auth.uid()
      OR EXISTS (
        SELECT 1 FROM public.diagram_shares ds
        WHERE ds.diagram_id = id AND ds.shared_with_id = auth.uid()
      )
      OR (
        workspace_id IS NOT NULL AND EXISTS (
          SELECT 1 FROM public.workspace_members wm
          WHERE wm.workspace_id = diagrams.workspace_id
            AND wm.user_id = auth.uid()
            AND wm.role IN ('owner', 'editor')
            AND wm.accepted_at IS NOT NULL
        )
      )
    )
  )
  WITH CHECK (
    -- Owner pode alterar qualquer campo
    (owner_id = auth.uid())
    OR
    -- Collaborator/editor: owner_id e title devem permanecer inalterados

      (
        EXISTS (
          SELECT 1 FROM public.diagram_shares ds
          WHERE ds.diagram_id = id AND ds.shared_with_id = auth.uid()
        )
        OR (
          workspace_id IS NOT NULL AND EXISTS (
            SELECT 1 FROM public.workspace_members wm
            WHERE wm.workspace_id = diagrams.workspace_id
              AND wm.user_id = auth.uid()
              AND wm.role IN ('owner', 'editor')
              AND wm.accepted_at IS NOT NULL
          )
        )
      )
      AND owner_id = public.get_diagram_owner(id)
      AND title = public.get_diagram_title(id)
    )
  );

Copyimport { describe, it, expect } from 'vitest';
import { exportToMermaid } from './exportService';
import type { DiagramNode, DiagramEdge } from '@/types/diagram';

function makeNode(id: string, type: string, label: string): DiagramNode {
  return { id, type, position: { x: 0, y: 0 }, data: { label, type } } as DiagramNode;
}

function makeEdge(id: string, source: string, target: string, protocol?: string): DiagramEdge {
  return { id, source, target, data: protocol ? { protocol } : {} } as DiagramEdge;
}

describe('exportToMermaid', () => {
  it('generates graph LR header', () => {
    const result = exportToMermaid([makeNode('n1', 'service', 'Svc')], []);
    expect(result.startsWith('graph LR')).toBe(true);
  });

  it('creates correct node shapes', () => {
    const nodes = [
      makeNode('n1', 'service', 'API'),
      makeNode('n2', 'database', 'DB'),
      makeNode('n3', 'queue', 'MQ'),
      makeNode('n4', 'external', 'Ext'),
    ];
    const result = exportToMermaid(nodes, []);
    expect(result).toContain('N0["API"]');
    expect(result).toContain('N1[("DB")]');
    expect(result).toContain('N2>"MQ"]');
    expect(result).toContain('N3("Ext")');
  });

  it('renders edges with protocol labels', () => {
    const nodes = [makeNode('n1', 'service', 'A'), makeNode('n2', 'service', 'B')];
    const edges = [makeEdge('e1', 'n1', 'n2', 'gRPC')];
    const result = exportToMermaid(nodes, edges);
    expect(result).toContain('N0 -->|gRPC| N1');
  });

  it('renders edges without labels', () => {
    const nodes = [makeNode('n1', 'service', 'A'), makeNode('n2', 'service', 'B')];
    const edges = [makeEdge('e1', 'n1', 'n2')];
    const result = exportToMermaid(nodes, edges);
    expect(result).toContain('N0 --> N1');
  });

  it('escapes special mermaid characters in labels', () => {
    const nodes = [makeNode('n1', 'service', 'Auth [v2] (beta)')];
    const result = exportToMermaid(nodes, []);
    expect(result).not.toContain('[v2]');
    expect(result).toContain('Auth  v2   beta');
  });
});
