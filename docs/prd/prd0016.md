Situação atual:

A policy diagrams_select_policy contém a cláusula:

Isso permite que qualquer usuário autenticado leia todos os diagramas compartilhados do sistema inteiro, não apenas aqueles onde é destinatário via diagram_shares. 3.5 R5 — Duplicação de lógica de snap entre onNodeDrag e onNodeDragStop
Arquivo afetado: src/hooks/useSnapGuides.ts

Situação atual:

Correção esperada:

Extrair uma função pura calculateSnap e reutilizar em ambos os handlers:

Copyinterface SnapResult {
  snapX: number | null;
  snapY: number | null;
  guides: GuideLine[];
}

function calculateSnap(
  draggedNode: Node,
  allNodes: Node[],
  threshold: number = SNAP_THRESHOLD,
): SnapResult {
  const guides: GuideLine[] = [];
  const dragX = draggedNode.position.x;
  const dragY = draggedNode.position.y;
  const dragW = draggedNode.measured?.width ?? 160;
  const dragH = draggedNode.measured?.height ?? 80;
  const dragCX = dragX + dragW / 2;
  const dragCY = dragY + dragH / 2;

  let snapX: number | null = null;
  let snapY: number | null = null;
  let bestDx = threshold;
  let bestDy = threshold;

  for (const node of allNodes) {
    if (node.id === draggedNode.id) continue;
    const nW = node.measured?.width ?? 160;
    const nH = node.measured?.height ?? 80;
    const nCX = node.position.x + nW / 2;
    const nCY = node.position.y + nH / 2;

    // Center X
    const dCX = Math.abs(dragCX - nCX);
    if (dCX < bestDx) {
      bestDx = dCX;
      snapX = nCX - dragW / 2;
      guides.push({ type: 'vertical', pos: nCX });
    }
    // Left edge
    const dLX = Math.abs(dragX - node.position.x);
    if (dLX < bestDx) {
      bestDx = dLX;
      snapX = node.position.x;
      guides.push({ type: 'vertical', pos: node.position.x });
    }
    // Right edge
    const dRX = Math.abs(dragX + dragW - (node.position.x + nW));
    if (dRX < bestDx) {
      bestDx = dRX;
      snapX = node.position.x + nW - dragW;
      guides.push({ type: 'vertical', pos: node.position.x + nW });
    }

    // Center Y
    const dCY2 = Math.abs(dragCY - nCY);
    if (dCY2 < bestDy) {
      bestDy = dCY2;
      snapY = nCY - dragH / 2;
      guides.push({ type: 'horizontal', pos: nCY });
    }
    // Top edge
    const dTY = Math.abs(dragY - node.position.y);
    if (dTY < bestDy) {
      bestDy = dTY;
      snapY = node.position.y;
      guides.push({ type: 'horizontal', pos: node.position.y });
    }
    // Bottom edge
    const dBY = Math.abs(dragY + dragH - (node.position.y + nH));
    if (dBY < bestDy) {
      bestDy = dBY;
      snapY = node.position.y + nH - dragH;
      guides.push({ type: 'horizontal', pos: node.position.y + nH });
    }
  }

  return { snapX, snapY, guides };
}
Copy
Os handlers onNodeDrag e onNodeDragStop devem chamar calculateSnap(draggedNode, nodes) e usar o resultado. O onNodeDrag define as guide lines e aplica snap; o onNodeDragStop limpa as guide lines e aplica snap final.
