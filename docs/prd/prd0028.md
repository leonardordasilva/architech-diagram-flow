FASE 1 — TYPE SAFETY: Utilitário getErrorMessage (+1 arquivo, ~33 edições)
Tarefa F1-T1: Criar utilitário getErrorMessage e migrar todos os catch blocks
Pilar: Type Safety & Error Handling Prioridade: Alta (Bloqueador) Referência: Code Review #1

Problema
33 ocorrências de `catch (err: any)` seguidas de `err.message` estão espalhadas por 12 arquivos de produção. Se `err` não for uma instância de `Error` (e.g., string, null, objeto de rede), `err.message` será `undefined` e pode crashar ou vazar valores inesperados na UI. O padrão `any` também desabilita o type-checking do TypeScript nesses blocos.

O que fazer
Passo 1 — Criar src/utils/getErrorMessage.ts:

```typescript
/**
 * Safely extracts a human-readable message from an unknown error value.
 * Use in every catch block instead of `(err: any) => err.message`.
 * @ref PRD-0028 F1-T1
 */
export function getErrorMessage(err: unknown): string {
  if (err instanceof Error) return err.message;
  if (typeof err === 'string') return err;
  if (err && typeof err === 'object' && 'message' in err && typeof (err as any).message === 'string') {
    return (err as { message: string }).message;
  }
  return 'Unknown error';
}
```

```typescript
import { useState, useCallback } from 'react';

export type ModalName =
  | 'upgrade'
  | 'billing'
  | 'account'
  | 'myDiagrams'
  | 'signOutConfirm'
  | 'shareDiagram'
  | 'import';

interface ModalManagerReturn {
  /** Check if a specific modal is open */
  isOpen: (name: ModalName) => boolean;
  /** Open a specific modal */
  open: (name: ModalName) => void;
  /** Close a specific modal */
  close: (name: ModalName) => void;
  /** Toggle a specific modal */
  toggle: (name: ModalName) => void;
  /** Set open state for a specific modal (for dialog onOpenChange) */
  setOpen: (name: ModalName, open: boolean) => void;
}

/**
 * Manages multiple modal states in a single hook to reduce
 * useState boilerplate in DiagramCanvas.
 * @ref PRD-0028 F4-T1
 */
export function useModalManager(): ModalManagerReturn {
  const [openModals, setOpenModals] = useState<Set<ModalName>>(new Set());

  const isOpen = useCallback((name: ModalName) => openModals.has(name), [openModals]);

  const open = useCallback((name: ModalName) => {
    setOpenModals((prev) => {
      const next = new Set(prev);
      next.add(name);
      return next;
    });
  }, []);

  const close = useCallback((name: ModalName) => {
    setOpenModals((prev) => {
      const next = new Set(prev);
      next.delete(name);
      return next;
    });
  }, []);

  const toggle = useCallback((name: ModalName) => {
    setOpenModals((prev) => {
      const next = new Set(prev);
      if (next.has(name)) next.delete(name);
      else next.add(name);
      return next;
    });
  }, []);

  const setOpen = useCallback((name: ModalName, value: boolean) => {
    if (value) open(name);
    else close(name);
  }, [open, close]);

  return { isOpen, open, close, toggle, setOpen };
}
```

```typescript
import { useAuth } from '@/hooks/useAuth';
import { usePlanStore } from '@/store/planStore';
import { useDiagramStore } from '@/store/diagramStore';
import { useQuery } from '@tanstack/react-query';
import { supabase } from '@/integrations/supabase/client';

/**
 * Returns whether toolbar actions should be locked.
 * Locked when: user is logged in + has unsaved canvas + reached diagram limit.
 * @ref PRD-0028 F4-T1
 */
export function useToolbarLocked(): boolean {
  const { user } = useAuth();
  const limits = usePlanStore((s) => s.limits);
  const currentDiagramId = useDiagramStore((s) => s.currentDiagramId);

  const { data: diagramCount = 0 } = useQuery({
    queryKey: ['diagram-count', user?.id],
    queryFn: async () => {
      if (!user) return 0;
      const { count } = await supabase
        .from('diagrams')
        .select('id', { count: 'exact', head: true })
        .eq('owner_id', user.id)
        .is('deleted_at', null);
      return count ?? 0;
    },
    enabled: !!user,
    staleTime: 30_000,
  });

  if (!user) return false;
  if (currentDiagramId) return false; // Has a saved diagram open
  if (limits.maxDiagrams === null) return false; // Unlimited
  return diagramCount >= limits.maxDiagrams;
}
```
